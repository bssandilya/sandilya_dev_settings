### script to setup proxy
function setproxy()
{
    export {http,https,ftp}_proxy='http://username:passwd@proxy:80'
}

function unsetproxy()
{
    unset {http,https,ftp}_proxy
}

# other useful functions
function mcd ()
{
    mkdir -p "$1" && cd "$1";
}

function wifipassword ()
{
    WIFINETWORK=$1
    if [ -z $WIFINETWORK ]; then
        echo "ERROR: Wifi Network name missing"
    else
        security find-generic-password -wa $WIFINETWORK
    fi
}
